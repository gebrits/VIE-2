VIE2.Util.isCurie = function (str) {

VIE2.Util.isLiteral = function (str) {
    try {
        $.rdf.resource(str, {namespaces: VIE2.namespaces});
        return false;
    } catch (e) {
        try {
            $.rdf.blank(str, {namespaces: VIE2.namespaces});
            return false;
        } catch (f) {
            try {
                $.rdf.literal(str, {namespaces: VIE2.namespaces});
                return true;
            } catch (g) {
                return false;
            }
        }
    }
};
